AlertOptions (todas las propiedades) 

const options = {
  message:           'string',           // Contenido del diálogo (texto o HTML si asHtml=true)
  title:             'string',           // Título en la cabecera (vacío = sin header)
  subTitle:          'string',           // Subtítulo bajo el título
  icon:              'string',           // Nombre icono lucide: 'info','check','warning','circle-x','question'...
  size:              'sm'|'md'|'lg'|'xl'|'fullscreen'|'none'|'image',
  showFooter:        false,              // Muestra barra de botones inferior
  showConfirmButton: false,              // Muestra botón Confirmar (además de Cerrar)
  disableClose:      false,              // Impide cerrar con Escape, backdrop o botón X
  asHtml:            false,              // Interpreta message como HTML
  autoCloseMs:       0,                 // Auto-cierre tras N ms (0 = desactivado)
  literals:          ['Cerrar', 'Aceptar'], // Textos de [cancelar, confirmar]
  onClose:           () => void,          // Callback al cerrarse (por cualquier vía)
  onCancel:          () => void,          // Callback al pulsar Cancelar / botón Cerrar
  onConfirm:         (instance) => bool|void, // Callback Confirmar. Return false impide cierre
  onAfterOpen:       (instance) => void,  // Callback tras montar en el DOM (via _show interno)
};

// AlertRef (retornado por showInfo, showSuccess, showDialog, etc.)
const ref = {
  instance,                    // La instancia de AlertComponent
  afterOpen: (cb) => void,   // Registra callback post-apertura
};

// Métodos de la instancia (AlertComponent):
ref.instance.close();          // Cierra el diálogo
ref.instance.canClose();       // Devuelve true si no está bloqueado
ref.instance.getContainer();   // Devuelve el HTMLElement del diálogo
ref.instance.setFeedback(txt); // Muestra texto de feedback en el footer

showInfo(message, options?)

Icono info. Sin footer por defecto.

Ver código 
// Básico
ds.showInfo('Operación completada con éxito.');

// Con título
ds.showInfo('Los datos se han guardado correctamente.', {
  title: 'Información',
});

// Auto-cierre
ds.showInfo('Se cerrará solo en 2 segundos.', {
  title: 'Auto-cierre',
  autoCloseMs: 2000,
});

// TODAS las opciones
ds.showInfo('Mensaje con todas las opciones posibles.', {
  title: 'Título completo',
  subTitle: 'Subtítulo descriptivo',
  size: 'md',
  showFooter: true,
  showConfirmButton: true,
  disableClose: false,
  autoCloseMs: 0,
  literals: ['Descartar', 'Entendido'],
  onClose:   () => console.log('onClose'),
  onCancel:  () => console.log('onCancel'),
  onConfirm: (instance) => console.log('onConfirm', instance),
});

showQuestion(message, options?)

Icono question. Footer + botón confirmar por defecto.

Ver código 
ds.showQuestion('¿Está seguro de que desea continuar?');

ds.showQuestion('¿Desea eliminar este registro?', {
  title: 'Confirmar eliminación',
  literals: ['Cancelar', 'Eliminar'],
  onConfirm: () => console.log('Eliminado'),
  onCancel:  () => console.log('Cancelado'),
});

// onConfirm retorna false → el diálogo NO se cierra
ds.showQuestion('Imposible confirmar esta demo.', {
  title: 'Confirmación condicionada',
  literals: ['Cancelar', 'Confirmar'],
  onConfirm: (instance) => {
    instance.setFeedback('Error de validación');
    return false;
  },
});

confirm(message, options?) → Promise<boolean>

Versión async/await. Resuelve true o false.

Ver código 
// Básico
const result = await ds.confirm('¿Guardar cambios?');
console.log(result); // true o false

// Con opciones
const ok = await ds.confirm('¿Exportar datos a CSV?', {
  title: 'Exportar datos',
  subTitle: 'Se generará un archivo con todos los registros',
  literals: ['No exportar', 'Exportar'],
  size: 'sm',
});
if (ok) ds.showSuccess('Exportación iniciada.', { autoCloseMs: 1500 });

showDialog(options?) – Genérico

Sin icono, con footer. Máxima personalización.

Ver código 
// Genérico básico
ds.showDialog({
  message: 'Diálogo genérico sin icono.',
  title: 'Diálogo base',
});

// HTML con formulario
ds.showDialog({
  title: 'Contenido HTML',
  asHtml: true,
  message: '<div>...inputs...</div>',
  size: 'md',
  showConfirmButton: true,
  literals: ['Cancelar', 'Enviar'],
  onConfirm: () => console.log('Enviado'),
});

// Fullscreen
ds.showDialog({
  title: 'Diálogo fullscreen',
  size: 'fullscreen',
  asHtml: true,
  message: '<div class="flex items-center justify-center h-full">...</div>',
});

// afterOpen — inyectar contenido dinámico
const ref = ds.showDialog({
  title: 'afterOpen callback',
  message: '<div id="dynamic">Cargando...</div>',
  asHtml: true,
});
ref.afterOpen((instance) => {
  const el = instance.getContainer()?.querySelector('#dynamic');
  if (el) el.textContent = '¡Inyectado desde afterOpen!';
});

// setFeedback — mostrar errores en el footer
ds.showDialog({
  title: 'setFeedback',
  message: 'Pulsa Confirmar para ver feedback.',
  showConfirmButton: true,
  onConfirm: (instance) => {
    instance.setFeedback('Error de validación');
    return false; // no cerrar
  },
});

showLoading(message, options?)

Barra de progreso animada. disableClose: true por defecto.

Ver código 
const ref = ds.showLoading('Procesando datos...');
setTimeout(() => ref.instance.close(), 3000);

const ref2 = ds.showLoading('Subiendo archivo...', { title: 'Subida' });
setTimeout(() => {
  ref2.instance.close();
  ds.showSuccess('Subido correctamente.', { autoCloseMs: 2000 });
}, 3000);

Otros: progressBar, close(), forceClose(), diálogos apilados

Ver código 
// Barra de progreso global (vía pubsub)
ds.showProgressBar(3000);

// Diálogos apilados con z-index incremental
ds.showInfo('Diálogo 1', { title: 'Capa 1' });
setTimeout(() => ds.showWarning('Diálogo 2', { title: 'Capa 2' }), 500);
setTimeout(() => ds.showError('Diálogo 3', {
  title: 'Capa 3',
  showConfirmButton: true,
  literals: ['Cerrar este', 'Cerrar TODOS'],
  onConfirm: () => ds.forceClose(),
}), 1000);

// Cerrar el último
ds.close();

// Cerrar todos
ds.forceClose();
